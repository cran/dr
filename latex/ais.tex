\Header{ais}{Australian athletes data}
\alias{AIS}{ais}
\keyword{datasets}{ais}
\begin{Description}\relax
Data on 102 male and 100 female athletes collected at the Australian
Institute of Sport, courtesy of Richard Telford and Ross Cunningham.
\end{Description}
\begin{Format}\relax
This data frame contains the following columns:
\describe{
\item[Sex] (0 = male or 1 = female)

\item[Ht] height (cm)

\item[Wt] weight (kg)

\item[LBM] lean body mass

\item[RCC] red cell count

\item[WCC] white cell count

\item[Hc] Hematocrit

\item[Hg] Hemoglobin

}
\end{Format}
